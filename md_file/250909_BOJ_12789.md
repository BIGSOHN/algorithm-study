**오늘의 메모(선택)**: 주기적으로 스택을 넣기 전 stack의 top이 번호와 일치하는 경우를 생각 못해서 애를 먹었다.

**코드(선택)**:

```c++
#include <iostream>
#include <stack>
#include <vector>

using namespace std;

int main() {
	int N, num;
	int k = 1;

	cin >> N;
	vector<int> v(N);
	stack<int> stk;
	for (int i = 0; i < N; i++) {
		cin >> v[i];
	}
	for (int i = 0; i < v.size(); i++) {
		if (v[i] == k) {
			k++;
			while (!stk.empty() && stk.top() == k) {
				stk.pop();
				k++;
			}
		} else {
			stk.push(v[i]);
		}
	}
	while (!stk.empty() && stk.top() == k) {
		stk.pop();
		k++;
	}
	if (!stk.empty()) {
		cout << "Sad" << endl;
	} else {
		cout << "Nice" << endl;
	}

	return (0);
}
```
