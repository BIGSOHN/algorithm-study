**문제 이름 또는 번호**: 1010 / 다리 놓기

**문제 링크**: https://www.acmicpc.net/problem/1010

**오늘의 메모(선택)**:

**코드(선택)**:

```c++
#include <iostream>

using namespace std;

long long combination(int m, int n) {
	if (n > m - n) {
		n = m - n;
	}
	long long result = 1;
	for (int i = 0; i < n; i++) {
		result = result * (m - i) / (i + 1);
	}
	return (result);
}

int main() {
	int num, N, M;
	cin >> num;
	for (int i = 0; i < num; i++) {
		cin >> N >> M;
		cout << combination(M, N) << '\n';
	}
	return (0);
}
```
