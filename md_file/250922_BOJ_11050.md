**오늘의 메모(선택)**: 이항계수 공식을 몰라서 인터넷을 참고했다.

참고 :

**코드(선택)**:

```c++
#include <iostream>

using namespace std;

long long factorial(int k) {
	long long result = 1;
	for (int i = k; i > 0; i--) {
		result *= i;
	}
	return (result);
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	int N, M;
	cin >> N >> M;
	long long result = factorial(N) / (factorial(M) * factorial(N - M));
	cout << result << '\n';

	return (0);
}
```
